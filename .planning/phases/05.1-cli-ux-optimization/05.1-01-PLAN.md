    **Zod Schema structure (per D-12):**
    ```typescript
    import { z } from "zod";

    const ConfigSchema = z.object({
      llm: z.object({
        timeoutMs: z.number().int().min(1000).max(30000).default(8000),
        maxRetries: z.number().int().min(0).max(5).default(2),
        parallelLimit: z.number().int().min(1).max(10).default(3),
      }),
      ui: z.object({
        spinnerEnabled: z.boolean().default(true),
        spinnerType: z.enum(["dots", "line", "moon"]).default("dots"),
        defaultLimit: z.number().int().min(1).max(100).default(10),
      }),
    });

    export type CliConfig = z.infer<typeof ConfigSchema>;
    ```

    **Environment variable mapping (per D-13):**
    - `SEEKU_LLM_TIMEOUT` -> `llm.timeoutMs`
    - `SEEKU_LLM_RETRIES` -> `llm.maxRetries`
    - `SEEKU_PARALLEL_LIMIT` -> `llm.parallelLimit`
    - `SEEKU_SPINNER_ENABLED` -> `ui.spinnerEnabled` (parse "true"/"false")
    - `SEEKU_DEFAULT_LIMIT` -> `ui.defaultLimit`

    **Fast-fail validation (per D-14):**
    ```typescript
    export function getConfig(): CliConfig {
      const raw = {
        llm: {
          timeoutMs: process.env.SEEKU_LLM_TIMEOUT ? parseInt(process.env.SEEKU_LLM_TIMEOUT, 10) : undefined,
          maxRetries: process.env.SEEKU_LLM_RETRIES ? parseInt(process.env.SEEKU_LLM_RETRIES, 10) : undefined,
          parallelLimit: process.env.SEEKU_PARALLEL_LIMIT ? parseInt(process.env.SEEKU_PARALLEL_LIMIT, 10) : undefined,
        },
        ui: {
          spinnerEnabled: process.env.SEEKU_SPINNER_ENABLED === "false" ? false : undefined,
          defaultLimit: process.env.SEEKU_DEFAULT_LIMIT ? parseInt(process.env.SEEKU_DEFAULT_LIMIT, 10) : undefined,
        },
      };

      const result = ConfigSchema.safeParse(raw);
      if (!result.success) {
        console.error("Invalid CLI configuration:", result.error.issues);
        throw new Error("CLI configuration validation failed. Check environment variables.");
      }
      return result.data;
    }
    ```

    Export both `getConfig` and `ConfigSchema` for type access.
