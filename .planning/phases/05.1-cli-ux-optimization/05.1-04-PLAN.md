    ```typescript
    /**
     * Execute promises with concurrency control using factory functions
     * Per D-08: MUST receive (() => Promise<T>)[] not Promise<T>[] to prevent premature execution
     * Per D-09: Default concurrency from config, typically 3
     */
    export async function promisePool<T>(
      tasks: (() => Promise<T>)[],  // Factory functions! NOT Promise<T>[]
      concurrency: number,
      signal?: AbortSignal
    ): Promise<T[]> {
      if (signal?.aborted) {
        throw new DOMException("Aborted", "AbortError");
      }

      const results: T[] = [];
      const executing: Promise<void>[] = [];

      for (const task of tasks) {
        // Check for abort before starting each task
        if (signal?.aborted) {
          break;  // Stop starting new tasks
        }

        // Execute task factory function to create promise
        const p = task().then(result => {
          results.push(result);
          // Remove from executing array when done
          const idx = executing.indexOf(p);
          if (idx !== -1) executing.splice(idx, 1);
        }).catch(err => {
          // Remove from executing on error too
          const idx = executing.indexOf(p);
          if (idx !== -1) executing.splice(idx, 1);
          throw err;  // Propagate error
        });

        executing.push(p);

        // Wait if at concurrency limit
        if (executing.length >= concurrency) {
          await Promise.race(executing);
        }
      }

      // Wait for remaining executing tasks
      await Promise.all(executing);

      return results;
    }
    ```

    **Key design points:**
    - `tasks` is array of factory functions `(() => Promise<T>)[]` per D-08
    - Factory functions are called inside loop, not before - prevents premature execution
    - Supports AbortSignal for graceful cancellation per D-10
    - Results collected as tasks complete (completion order)
    - Concurrency controlled by `executing.length >= concurrency` check

    Export promisePool at module level.

    ```typescript
      candidates: HydratedCandidate[],
      conditions: SearchConditions,
      signal: AbortSignal
    ): Promise<void> {
      const config = getConfig();

      // Filter candidates without cached profiles
      const uncached = candidates.filter(c => {
        // Quick check without full cache lookup
        return true;  // Will be filtered in preload task
      });

      // D-09: Use config.llm.parallelLimit (default 3)
      const preloadTasks = uncached.map(candidate => async () => {
        if (signal.aborted) return;
    ```
