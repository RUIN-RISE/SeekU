    ```typescript
    private async showCandidateDetail(selected: HydratedCandidate, conditions: SearchConditions): Promise<void> {
      const config = getConfig();

      // D-01: Use ora spinner for loading indication
      const spinner = config.ui.spinnerEnabled
        ? ora({ text: `Loading deep insight for ${selected.name}...`, spinner: config.ui.spinnerType }).start()
        : null;

      try {
        let profile: MultiDimensionProfile | null = await this.cacheRepo.getProfile(selected.personId);
        const { person, evidence } = selected._hydrated;

        if (!profile) {
          // D-05: Phase-aware text updates
          if (spinner) spinner.text = "Cache check... (miss)";
          // Brief pause for user to see the phase

          if (spinner) spinner.text = "Calculating hybrid scores...";
          const rules = this.scorer.scoreByRules(person, evidence, conditions);
          const llm = await this.scorer.scoreByLLM(person, evidence);
          const experienceBonus = this.scorer.calculateExperienceMatch(person, evidence, conditions);
          profile = this.scorer.aggregate(rules, llm, experienceBonus);

          if (spinner) spinner.text = "Generating summary & highlights...";
          profile = await this.generator.generate(person, evidence, profile);

          await this.cacheRepo.setProfile(selected.personId, profile, profile.overallScore);

          if (spinner) spinner.succeed("Analysis complete");
        } else {
          // D-05: Cache hit feedback
          if (spinner) {
            spinner.text = "Cache check... (hit!)";
            spinner.succeed("Loaded from cache");
          }
        }

        // D-03: Stop spinner before any interactive prompts
        if (spinner && spinner.isSpinning) spinner.stop();

        console.log(this.renderer.renderProfile(person, evidence, profile));
        console.log(chalk.dim("Press Enter to return to list..."));
        await new Promise(resolve => process.stdin.once("data", resolve));

      } catch (e) {
        // D-01: Handle failure with spinner.fail()
        if (spinner) spinner.fail("Analysis failed");
        throw e;
      }
    }
    ```

    - The current code already returns to interactionLoop which calls tui.selectCandidate()
    - The "Press Enter to return to list" pause gives ora time to clean up stderr
