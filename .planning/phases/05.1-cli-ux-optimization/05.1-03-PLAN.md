    ```typescript
    import { getConfig } from "./config.js";

    /**
     * Determine if an error is retryable (transient) vs permanent (auth/validation)
     * Per D-16: 429/503/AbortError/ETIMEDOUT are retryable, 401/400 are NOT
     */
    export function isRetryable(error: Error): boolean {
      const message = error.message.toLowerCase();
      const name = error.name;

      // Transient errors - retry
      if (name === "AbortError") return true;
      if (message.includes("429")) return true;  // Rate limit
      if (message.includes("503")) return true;  // Service unavailable
      if (message.includes("etimedout")) return true;
      if (message.includes("econnreset")) return true;
      if (message.includes("econnrefused")) return true;
      if (message.includes("network")) return true;

      // Permanent errors - do NOT retry
      if (message.includes("401")) return false;  // Authentication
      if (message.includes("403")) return false;  // Forbidden
      if (message.includes("400")) return false;  // Validation/bad request

      // Default: retry unknown errors (conservative for CLI)
      return true;
    }
    ```

    **retryWithBackoff function (per D-17 and D-18):**
    ```typescript
    /**
     * Retry an async operation with exponential backoff
     * Per D-17: Delays are 1s, 2s, 4s (doubling each attempt)
     * Per D-18: Logs retry progress to console
     */
    export async function retryWithBackoff<T>(
      operation: () => Promise<T>,
      options?: {
        maxRetries?: number;
        operationName?: string;
        onRetry?: (attempt: number, error: Error) => void;
      }
    ): Promise<T> {
      const config = getConfig();
      const maxRetries = options?.maxRetries ?? config.llm.maxRetries;
      const operationName = options?.operationName ?? "operation";

      let lastError: Error | null = null;

      for (let attempt = 0; attempt <= maxRetries; attempt++) {
        try {
          return await operation();
        } catch (e) {
          const error = e instanceof Error ? e : new Error(String(e));
          lastError = error;

          if (!isRetryable(error)) {
            // Permanent error - don't retry
            console.error(`${operationName} failed permanently:`, error.message);
            throw error;
          }

          if (attempt < maxRetries) {
            // Exponential backoff: 1000, 2000, 4000 ms
            const delayMs = Math.pow(2, attempt) * 1000;
            console.warn(`${operationName} failed (attempt ${attempt + 1}/${maxRetries + 1}), retrying in ${delayMs}ms...`);

            if (options?.onRetry) {
              options.onRetry(attempt + 1, error);
            }

            await new Promise(resolve => setTimeout(resolve, delayMs));
          }
        }
      }

      // All retries exhausted
      console.error(`${operationName} failed after ${maxRetries + 1} attempts`);
      throw lastError || new Error("Retry failed with unknown error");
    }
    ```

    ```typescript
      const controller = new AbortController();

      try {
        // Sanitize external data...
        const prompt = `...`;

        // Wrap LLM call with retry (per D-15)
        const response = await retryWithBackoff(
          async () => {
            const timeoutId = setTimeout(() => controller.abort(), LLM_TIMEOUT_MS);
            try {
              const result = await this.llm.chat([
                { role: "system", content: "You are an expert technical recruiter..." },
                { role: "user", content: prompt }
              ], { signal: controller.signal });
              clearTimeout(timeoutId);
              return result;
            } catch (e) {
              clearTimeout(timeoutId);
              throw e;
            }
          },
          { operationName: "LLM scoring" }
        );

        // Parse response with safeParseJSON...
        const result = safeParseJSON(response.content, LLMScoresSchema, {...});

        return {
          projectDepth: result.data.projectDepth,
          academicImpact: result.data.academicImpact,
          communityReputation: result.data.communityReputationBoost
        };
      } catch (e) {
        // This catch handles errors after all retries exhausted
        if (e instanceof Error && e.name === "AbortError") {
          console.warn("LLM scoring timed out after all retries");
        } else if (!isRetryable(e instanceof Error ? e : new Error(String(e)))) {
          console.warn("LLM scoring failed permanently:", e instanceof Error ? e.message : String(e));
        }

        // Fallback with dynamic defaults
        const repoCount = evidence.filter(e => e.evidenceType === "repository").length;
        return {
          projectDepth: Math.min(80, 40 + repoCount * 5),
          academicImpact: 40,
          communityReputation: 5
        };
      }
    }
    ```
