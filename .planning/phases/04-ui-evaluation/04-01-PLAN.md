3. Implement the handler function:
   ```typescript
   async function handleProfile(db: SeekuDatabase, request: FastifyRequest, reply: FastifyReply) {
     const params = request.params as { personId: string };
     const person = await db.select().from(persons).where(
       and(eq(persons.id, params.personId), eq(persons.searchStatus, "active"))
     );
     if (!person.length) {
       return reply.status(404).send({ error: "not_found" });
     }
     const evidence = await db.select().from(evidenceItems).where(eq(evidenceItems.personId, params.personId));
     return { person: person[0], evidence };
   }
   ```

4. Export registerProfileRoutes function:
   ```typescript
   export function registerProfileRoutes(server: FastifyInstance, db: SeekuDatabase) {
     server.get("/profiles/:personId", async (request, reply) => handleProfile(db, request, reply));
   }
   ```

Do NOT add caching, rate limiting, or authentication - keep it minimal for MVP.
